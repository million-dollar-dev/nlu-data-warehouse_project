CREATE TABLE date_dim (
    id SERIAL PRIMARY KEY,                  -- ID duy nhất
    full_date DATE NOT NULL,                -- Ngày đầy đủ (ISO format: YYYY-MM-DD)
    day_of_month INTEGER NOT NULL,          -- Ngày trong tháng
    month INTEGER NOT NULL,                 -- Số tháng trong năm
    day_name TEXT NOT NULL,                 -- Tên ngày (Thứ)
    month_name TEXT NOT NULL,               -- Tên tháng
    year INTEGER NOT NULL,                  -- Năm
    start_of_week TEXT NOT NULL,            -- Tuần bắt đầu (chuỗi)
    day_of_week INTEGER NOT NULL,           -- Thứ trong tuần
    day_of_year INTEGER NOT NULL,           -- Ngày trong năm
    iso_week INTEGER NOT NULL,              -- Số tuần ISO trong năm
    iso_week_year TEXT NOT NULL,            -- Năm và tuần ISO (chuỗi)
    start_of_iso_week DATE NOT NULL,        -- Ngày bắt đầu tuần ISO
    iso_week_alt INTEGER,                   -- Tuần ISO phụ
    iso_week_year_alt TEXT,                 -- Năm và tuần ISO phụ (chuỗi)
    start_of_iso_alt DATE,                  -- Ngày bắt đầu tuần ISO phụ
    quarter TEXT NOT NULL,                  -- Quý hiện tại (chuỗi)
    quarter_num INTEGER NOT NULL,           -- Số quý
    holiday_flag TEXT NOT NULL,             -- Ngày lễ/không
    is_weekend TEXT NOT NULL                -- Loại ngày (Weekend/Weekday)
);
